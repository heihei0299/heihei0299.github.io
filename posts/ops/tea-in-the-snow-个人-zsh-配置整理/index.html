6. 自定义工具函数#

# yazi 文件管理器集成，退出时自动切换到选中目录
function yy() {
	local tmp="$(mktemp -t "yazi-cwd.XXXXXX")"
	yazi "$@" --cwd-file="$tmp"
	if cwd="$(cat -- "$tmp")" && [ -n "$cwd" ] && [ "$cwd" != "$PWD" ]; then
		builtin cd -- "$cwd"
	fi
	rm -f -- "$tmp"
}

输入yy启动yazi文件管理器，退出时会自动cd到你最后浏览的目录，非常实用。
