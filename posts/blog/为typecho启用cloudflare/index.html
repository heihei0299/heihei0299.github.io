为typecho启用cloudflare#

配置 nginx#

  1. 只允许特定的 sni 或者 host 访问特定的服务，其他未知 sni 或者 host 直接返回404.
  2. 只允许cloudflare的 ip 访问80端口
vim /etc/nginx/conf.d/default.conf


# 默认 server，匹配所有未定义的 Host
server {
    listen 80 default_server;
    server_name _;  # 匹配任意 Host
    include /etc/nginx/conf.d/cf-ip.conf;
    deny all;
    return 404;
}

vim /etc/nginx/conf.d/typecho.conf

server {
    listen          80;
    server_name     yourservername;    
    root            /var/www/html/typecho;
    index           index.html index.htm index.php;
    
    location / {
    try_files $uri $uri/ /index.php?$args;  
}
location ~ \.php$ {
    fastcgi_pass   127.0.0.1:5050;      
    fastcgi_index  index.php;
    fastcgi_param  SCRIPT_FILENAME  $document_root$fastcgi_script_name;
    include        fastcgi_params;
}
